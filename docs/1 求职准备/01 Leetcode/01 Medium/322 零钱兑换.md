class Solution {
public:

    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount + 1, amount + 1);
        dp[0] = 0;

        for(int val = 1; val <= amount; val++){
            for(int i = 0 ; i < coins.size(); i++){
                if(val - coins[i] >= 0){
                    dp[val] = min(dp[val], dp[val - coins[i]] + 1);
                }
            }
        }

        return dp[amount] > amount ? -1 : dp[amount];
    }
};
