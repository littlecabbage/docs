        vector<vector<int>> ans;
        sort(intervals.begin(), intervals.end());
        int start = intervals[0][0], end = intervals[0][1];
        int new_start, new_end;

        int n = intervals.size();
        for(int i = 1; i < n;i++){
            new_start = intervals[i][0];
            new_end = intervals[i][1];
