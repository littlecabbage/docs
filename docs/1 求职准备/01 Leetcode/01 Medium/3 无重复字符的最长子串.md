class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> buffer;
        int n = s.size();

        int left, right;
        left = right = 0;

        int ans = 0;

        while(right < n){
            char ch = s[right];

            while(occ.count(ch)){
                occ.erase(s[left]);
                left++;
            }

            occ.insert(ch);
            right++;

            ans = max(ans, right - left);
        }

        return ans;
    }
};
