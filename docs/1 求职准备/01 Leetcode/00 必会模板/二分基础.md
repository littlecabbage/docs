# 1. 等于 x 的下标
![[Pasted image 20220606111248.png]]


```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1, mid;

        while(left <= right){
            mid = (left + right) / 2;
            if(nums[mid] == target){
                return mid;
            }

            else if(nums[mid] < target){
                left = mid + 1;
            }else{
                right = mid - 1;
            }
        }

        return -1;
    }
};
```

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0, right = nums.size()-1;

        int ans = -1;
        while(left <= right){
            int mid = (left + right) / 2;

            if(target <= nums[mid]){
                ans = mid;
                right = mid - 1;
            }else{
                left = mid + 1;
            }
        }

        return ans;
    }
};
```

```cpp
class Solution {
public:
    int mySqrt(int x) {
        long long left = 0, right = 10000000;
        long long ans = 0;
        while(left <= right){
            long long mid = (left + right) / 2;

            if(x == mid * mid){
                return mid;
            }else if(x < mid*mid){
                right = mid -1;;
            }else{
                ans = mid;
                left = mid + 1;
            }
        }
        return ans;
    }
};
```
