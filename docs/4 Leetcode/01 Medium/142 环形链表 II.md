/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if(head == NULL || head->next == NULL) return NULL;
        int N = 0;
        ListNode *fast, *slow;
        slow = head;
        fast = head->next;
        while(fast != slow){
            slow = slow -> next;
            N++;

            if(fast && fast->next) fast = fast -> next -> next;
            else break;
        }

        if(fast == NULL || fast->next == NULL) return NULL;
        if(fast == slow){
            slow = head;
            fast = head -> next;
            while(N--){
                fast = fast->next;
            }
            while(fast != slow){
                fast = fast->next;
                slow = slow->next;
            }
            return slow;
        }

        return NULL;
    }
};
