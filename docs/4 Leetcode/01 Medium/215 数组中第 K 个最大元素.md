class Solution {
public:
    void build_heap(vector<int> & heap){
        int len = heap.size();
        len /= 2;
        while(len--){
            int left = len*2 + 1;
            int right = len*2 + 2;
            if(left < heap.size() && heap[left] < heap[len]) swap(heap[left], heap[len]);
            if(right < heap.size() && heap[right] < heap[len]) swap(heap[right], heap[len]);
        }
    }

    int findKthLargest(vector<int>& nums, int k) {
        vector<int> min_heap(k, 0);
        for(int i = 0; i < k; i++){
            min_heap[i] = nums[i];
        }

        build_heap(min_heap);
        for(int i = k; i < nums.size(); i++){
            if(nums[i] > min_heap[0]){
                //min_heap.push_back(nums[i]);
                min_heap[0] = nums[i];
                //min_heap[k-1] = nums[i];
                build_heap(min_heap);
            }
        }

        return min_heap[0];
    }
};
