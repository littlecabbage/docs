两趟扫描，哈希表

```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int> set;
        for(int i = 0; i < nums.size(); i++){
            set.insert(nums[i]);
        }

        int cnt = 0;
        for(int i = 0; i < nums.size(); i++){
            if(set.find(nums[i]-1) == set.end()){
                int start = nums[i];
                int end = nums[i];
                while(set.find(end) != set.end()){
                    end++;
                }

                cnt = max(cnt, end - start);
            }
        }

        return cnt;
    }
};
```
